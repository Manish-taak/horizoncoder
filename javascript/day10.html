Day 10

Asynchronous Programming in JavaScript

1. Callbacks and Callback Hell

A callback is a function passed as an argument to another function to be executed later, often after an asynchronous task is complete. Callback Hell occurs when multiple callbacks are nested, making the code difficult to read and maintain.

                
function fetchData(callback) {
    setTimeout(() => {
        document.getElementById('callbackOutput').innerText += "Data fetched from server.\n";
        callback();
    }, 1000);
}

function processData(callback) {
    setTimeout(() => {
        document.getElementById('callbackOutput').innerText += "Data processed.\n";
        callback();
    }, 1000);
}

function saveData(callback) {
    setTimeout(() => {
        document.getElementById('callbackOutput').innerText += "Data saved.\n";
        callback();
    }, 1000);
}

// Example of Callback Hell
function executeCallbackHell() {
    fetchData(function() {
        processData(function() {
            saveData(function() {
                document.getElementById('callbackOutput').innerText += "All tasks completed.";
            });
        });
    });
}

2. Promises and Promise Chaining

A Promise is an object that represents the eventual completion (or failure) of an asynchronous operation. It helps avoid callback hell by allowing you to chain asynchronous calls in a more readable way.

                
function fetchDataPromise() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            document.getElementById('promiseOutput').innerText += "Data fetched from server.\n";
            resolve();
        }, 1000);
    });
}

function processDataPromise() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            document.getElementById('promiseOutput').innerText += "Data processed.\n";
            resolve();
        }, 1000);
    });
}

function saveDataPromise() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            document.getElementById('promiseOutput').innerText += "Data saved.\n";
            resolve();
        }, 1000);
    });
}

// Example of Promise Chaining
function executePromiseChaining() {
    fetchDataPromise()
        .then(processDataPromise)
        .then(saveDataPromise)
        .then(() => {
            document.getElementById('promiseOutput').innerText += "All tasks completed.";
        })
        .catch((error) => {
            document.getElementById('promiseOutput').innerText += "An error occurred: " + error;
        });
}
