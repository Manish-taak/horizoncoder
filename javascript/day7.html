String Methods

Learn more about String Methods on W3Schools 
                
let greeting = "Hello, World!";
console.log(greeting.toUpperCase()); // Converts to uppercase
console.log(greeting.replace("World", "JavaScript")); // Replaces "World" with "JavaScript"

Array Methods

Learn more about Array Methods on W3Schools 
                
let fruits = ["Apple", "Banana"];
fruits.push("Mango"); // Adds "Mango" to the array
console.log(fruits);
fruits.pop(); // Removes the last element
console.log(fruits);

Math Methods

Learn more about Math Methods on W3Schools 
                
console.log(Math.PI); // Prints the value of PI
console.log(Math.sqrt(16)); // Prints the square root of 16

Object Methods

Learn more about Object Methods on W3Schools 
                
let car = { brand: "Toyota", model: "Camry" };
console.log(car.brand); // Accesses the "brand" property
car.model = "Corolla"; // Modifies the "model" property
console.log(car);

Date Methods

Learn more about Date Methods on W3Schools 
                
let currentDate = new Date();

console.log(currentDate.getFullYear()); // Gets the year
