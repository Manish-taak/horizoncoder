Async/Await in JavaScript

1. Simplifying Asynchronous Code with Async/Await

Using async/await makes asynchronous code easier to read and maintain, avoiding the need for nested callbacks or complex promise chains.

                
async function fetchDataAsync() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            document.getElementById('asyncOutput').innerText += "Data fetched from server.\n";
            resolve();
        }, 1000);
    });
}

async function processDataAsync() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            document.getElementById('asyncOutput').innerText += "Data processed.\n";
            resolve();
        }, 1000);
    });
}

async function saveDataAsync() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            document.getElementById('asyncOutput').innerText += "Data saved.\n";
            resolve();
        }, 1000);
    });
}

// Example of Async/Await
async function executeAsyncAwait() {
    try {
        await fetchDataAsync();
        await processDataAsync();
        await saveDataAsync();
        document.getElementById('asyncOutput').innerText += "All tasks completed.";
    } catch (error) {
        document.getElementById('asyncOutput').innerText += "An error occurred: " + error;
    }
}

2. Error Handling in Async/Await

Using try...catch blocks to handle errors in asynchronous operations is simple and effective with async/await.

                
async function fetchDataWithError() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            document.getElementById('errorOutput').innerText += "Data fetching failed.\n";
            reject("Network error");
        }, 1000);
    });
}

// Example of Async/Await with Error Handling
async function executeAsyncAwaitWithError() {
    try {
        await fetchDataWithError();
    } catch (error) {
        document.getElementById('errorOutput').innerText += "Caught an error: " + error;
    }
}
